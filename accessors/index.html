const complex = {
  r: 10,
  alpha: 3.14, 
  get x(): number {
    return this.r * Math.sin(this.alpha); 
  },
  set y(newY: number) { 
    const currentX = this.x;
    this.r = Math.sqrt(currentX * currentX + newY * newY);
    this.alpha = Math.atan2(newY, currentX);
  }
}

complex.x = () => 0;
