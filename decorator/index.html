Decorators

Declaration 

function Entity(target: T) {
    doSomethingWith(target);
    return target;
}

function Loggable(metadata: string) {
  return function(instance) {
    doSomethingWith(instance);
  }
}

Usage 

@Entity
class Person {
  @JsonSerializable('userName')
  name: string;

  @PartiallyHidden public @Loggable('Said hi') SayHi(@Validate greetings: string) {

  }
}
