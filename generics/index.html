Generics

Generic class

class Container<T> {
  private content: T;

  public set(newElement: T) {
    this.content = newElement;
  }

  public get(): T {
    return this.content
  }
}

Generic function

function swop<T>(tuple: [T, T]): [T, T] {
    return [tuple[1], tuple[0]];
}

Generic functional type

let mySwop: <T>(tuple: [T, T]) => [T, T] = swop;
