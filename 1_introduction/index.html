Arrays

const squares = [1,2,4,9];
const propertyDefinition = ["height", 200];
const zoo = ["Zebra", 123, false];
 const squares: Array<number> = [1,2,4,9]; //Regular array
const propertyDefinition: [string, number] = ["height", 200]; //Tuple
const zoo: Array<number|string|boolean> = ["Zebra", 123, false]; //Array of union
